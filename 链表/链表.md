
# 静态链表
```
#include<stdio.h>
#include<string.h>
#include<stdlib.h>
struct LianBiao
{
	int dates;
	struct LianBiao* nest;
};

void node()
{
	struct LianBiao node1 = { 10,NULL };
	struct LianBiao node2 = { 20,NULL };
	struct LianBiao node3 = { 30,NULL };
	struct LianBiao node4 = { 40,NULL };
	struct LianBiao node5 = { 50,NULL };
	node1.nest = &node2;
	node2.nest = &node3;
	node3.nest = &node4;
	node4.nest = &node5;
	struct LianBiao* Bianli = &node1;
	while (Bianli != NULL)
	{
		printf("%d   ", Bianli->dates);
		Bianli = Bianli->nest;
	}
}//链表两个域的设置     一个值域   一个指针域

int main()
{
	node();
	system("pause");
	return 0;
}
```
