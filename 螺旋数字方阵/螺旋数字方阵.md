**需要输入循环次数 之后输入方阵阶数**
```
#include<stdio.h>
#include<string.h>
int num[20][20];
int main()
{
	
    int N;
    scanf("%d",&N);
    while(N--)
    {
        int n;
        scanf("%d",&n);
        int i,j;
        int js=1;
        memset(num,0,sizeof(num));
        for(i=0; i<=n/2; i++)
        {
            for(j=i; j<=n-i-1; j++)
            {
                num[i][j]=js++;
            }
            //1 2 3 
            // 8   4
            //7  6  5
            for(j=i+1; j<=n-i-1; j++)
            {
                num[j][n-i-1]=js++;
            }
            
           for(j=n-i-2;j>=i;j--)
           {
               num[n-i-1][j]=js++;
           }
           
           for(j=n-i-2;j>i;j--)
           {
               num[j][i]=js++;
           }
        }
            for(i=0; i<n; i++)
            {
                for( j=0; j<n; j++)
                    printf("%5d",num[i][j]);
                printf("\n");
            }
    }
 
    return 0;
}
```
