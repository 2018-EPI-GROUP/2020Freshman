```
#include <stdio.h>
struct student
{
    char name[101];
    int score;
};
struct student a[1000010], t;
void quickSort(int left, int right)
{
    struct student temp = a[left];
    int i, j;
    if (left > right)
        return;
    // 6 24 4 65 8 10 
    //4 6 24 65 8 10
    //i = 0 +   = 1
    //j = 5- - - = 2 - = 1
    //
    temp = a[left]; //temp中存的就是基准数
    i = left;
    j = right;
    while (i != j)
    {
        while (a[j].score >= temp.score && i < j)
        {
            j--;
        }
   
        while (a[i].score <= temp.score && i < j)
        {
            i++;
        }
      
        if (i < j)
        {
            t = a[i];
            a[i] = a[j];
            a[j] = t;
        }
    }
    //将基准数归位
    a[left] = a[i];
    a[i] = temp;

    quickSort(left, i - 1);
    quickSort(i + 1, right);

}
int main()
{
    int i, j, n;
    scanf("%d", &n); //输入一个数n
    for (i = 1; i <= n; i++) //循环读入n个人名和分数
        scanf("%s %d", a[i].name, &a[i].score);
    //按分数从高到低进行排序
    quickSort(1, n);
    for (i = 1; i <= n; i++)//输出人名
        printf("%s ", a[i].name);
    return 0;
}
```
