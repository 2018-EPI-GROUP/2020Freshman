void XR(int* p, int len)
{
	int i;
	int j;
	int temp;
	int step = len / 2;
	while (step)
	{
		for (i = step; i < len; i++)
		{
			temp = p[i];
			j = i - step;
			while (j >= 0 && p[j] > temp)
			{
				p[j + step] = p[j];
				j -= step;
			}
			p[j + step] = temp;
		
		
		}

		step = step / 2;
	}

}
